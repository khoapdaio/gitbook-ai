class Solution {
public:
    string predictPartyVictory(string senate) {
        queue<int> radiant, dire;
        int n = senate.size();

        // ÄÆ°a index vÃ o queue

            else dire.push(dIndex + n);
        }

        return radiant.empty() ? "Dire" : "Radiant";
    }
