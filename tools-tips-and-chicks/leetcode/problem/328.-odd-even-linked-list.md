class Solution {

    ListNode* oddEvenList(ListNode* head) {
        if (!head || !head->next) return head;
        
        ListNode *odd = head, *even = head->next, *evenHead = even;
        
        while (even && even->next) {

        }
        
        odd->next = evenHead;
